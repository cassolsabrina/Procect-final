Data analysis

The analysis of data from the in vitro experiment was conducted as follows:

library(readxl)
library(tidyverse)
library(ggplot2)

fung <- read_excel("fung_myc.xlsx")

library(DT)
datatable(fung)
#exploratory analysis

fung2 <- fung |>
  group_by(fungicide, dose, time) |> 
  summarise(growth_mean = mean(growth))


pfung2 <- fung2 |> 
   ggplot(aes(dose, growth_mean, group = time,
              color = factor(time)))+
  geom_point()+
  geom_line()+
  facet_wrap(~fungicide) +
      scale_y_continuous(n.breaks = 5, limits = c(0,5)) +
      labs(x = "Dose (μg a.i. mL)", y = "Mycelial growth (cm)", title= "Effect of fungicides on mycelial growth", color = "Experiment" ) +
      theme_minimal()+
    theme(plot.title = element_text(hjust = 0.5))

dadosec <- data.frame(  Trat = c("Ridomil", "Collis", "Tutor", "Comet"),
  Estimate = c(22.00, 67.265, 427.910, 90.471),
  Lower = c(-105.908, -46.359, 367.992, 22.458),
  Upper = c(149.908, 180.890, 487.828, 158.483))

datatable(dadosec)
library(ggplot2)

plotEC <- dadosec |>
  ggplot(aes(Trat, Estimate)) +
  geom_point() +
  ylim(-150,600)+
  geom_errorbar(aes(ymin = Lower, ymax = Upper), width = 0.1) +  
   geom_text(aes(label = round(Estimate, 2)), hjust = 1.2) +  
  labs(x = "Fungicide", y = "EC50") + 
  theme_minimal()
